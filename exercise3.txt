// Розробка калькулятора з використанням prompt() та змінних

// Запитуємо користувача про перший номер
var firstNumber = prompt("Введіть перший номер:");

// Зберігаємо перший номер в змінну
var number1 = parseFloat(firstNumber);

// Запитуємо користувача про другий номер
var secondNumber = prompt("Введіть другий номер:");

// Зберігаємо другий номер в змінну
var number2 = parseFloat(secondNumber);

// Обчислюємо суму двох чисел та відображаємо результат за допомогою alert()
var sum = number1 + number2;
alert("Сума двох чисел: " + sum);

// Обчислення різниці двох чисел та відображення результату
var difference = number1 - number2;
alert("Різниця двох чисел: " + difference);

// Обчислення добутку двох чисел та відображення результату
var product = number1 * number2;
alert("Добуток двох чисел: " + product);

// Обчислення частки двох чисел та відображення результату
var quotient = number1 / number2;
alert("Частка двох чисел: " + quotient);
